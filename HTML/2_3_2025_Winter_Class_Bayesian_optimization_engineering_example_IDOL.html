##########################################################################################################
# Acquisition function  (Surrogate predictor)
def SP_function(x, sm):

    # Predict the mean and variance at point x
    y_pred = sm.predict_values(x) #  ()

    return y_pred[0]

##########################################################################################################
# Acquisition function  (Mean-squared error)
def MSE_function(x, sm):

    # Predict the mean and variance at point x
    y_s2 = sm.predict_variances(x) #   ()

    return y_s2[0]

##########################################################################################################
# Acquisition function  (Expected improvement)
def EI_function(x, sm, yt):
    y_min = np.min(yt) #    
    # Predict the mean and variance at point x
    y_pred = sm.predict_values(x) #  ()
    y_s2 = sm.predict_variances(x) #   ()

    # Avoid division by zero
    y_s = np.sqrt(y_s2) # 
    y_s = np.maximum(y_s, 1e-16) #     trick

    # EI calculation
    EI_value = (y_min - y_pred)*norm.cdf((y_min - y_pred)/y_s) + y_s * norm.pdf((y_min - y_pred)/y_s)

    return EI_value[0]

##########################################################################################################
# Acquisition function  (Lower-confidence bound)
def LCB_function(x, sm):

    # Predict the mean and variance at point x
    y_pred = sm.predict_values(x) #  ()
    y_s2 = sm.predict_variances(x) #   ()
    y_s = np.sqrt(y_s2) # 

    # LCB calculation
    A = 1; # User-defined parameter (Generally, 1~4)
    LCB_value = y_pred - A*y_s

    return LCB_value[0]
